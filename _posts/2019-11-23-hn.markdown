    select
        count(*) as num_stories,
        avg(score) as avg_score,
        format_timestamp("%A %H", time_ts) as time
    from `bigquery-public-data.hacker_news.stories`
    group by time
    order by avg_score desc;




    select
        count(score) as n,
        score
    from `bigquery-public-data.hacker_news.stories`
    where format_timestamp("%A %H", time_ts) = "Sunday 14"
    group by score
    order by score desc;
